509/**
510 * Save custom data from Pressbooks' Status & Visibility panel.
511 *
512 * @since 5.0.0
513 *
514 * @param int $post_id Post ID.
515 * @param \WP_Post $post Post object.
516 * @param bool $update Whether this is an existing post being updated or not.
517 */
518function publish_fields_save( $post_id, $post, $update ) {
519    // Sanity checks
520    if ( empty( $_POST ) ) { // @codingStandardsIgnoreLine
521        return;
522    }
523    global $pagenow;
524    if ( ! in_array( $pagenow, [ 'post.php', 'post-new.php' ], true ) ) {
525        return;
526    }
527    if ( ! in_array(
528        $post->post_type, [
529            'front-matter',
530            'back-matter',
531            'chapter',
532            'glossary',
533        ], true
534    ) ) {
535        return;
536    }
537    if ( wp_is_post_revision( $post_id ) ) {
538        return;
539    }
540    if ( $post->post_status === 'trash' ) {
541        return;
542    }
543
544    // Set a static variable to fix infinite hook loop
545    static $recursion = false;
546    if ( ! $recursion ) {
547        $recursion = true;
548
549        $show_in_glossary_lists = ( isset( $_POST['glossary_visibility'] ) && (int) $_POST['glossary_visibility'] === 1 ) ? true : false;
550        $show_in_web = ( isset( $_POST['web_visibility'] ) && (int) $_POST['web_visibility'] === 1 ) ? true : false;
551        $require_password = ( isset( $_POST['require_password'] ) && (int) $_POST['require_password'] === 1 ) ? true : false;
552        $show_in_exports = ( isset( $_POST['export_visibility'] ) && (int) $_POST['export_visibility'] === 1 ) ? true : false;
553        $show_title = ( isset( $_POST['pb_show_title'] ) && $_POST['pb_show_title'] === 'on' ) ? 'on' : false;
554
555        // Content Visibility
556        if ( $post->post_type === 'glossary' ) {
557            // Glossary
558            $post_status = $show_in_glossary_lists ? 'publish' : 'private';
559        } else {
560            // Every other post_type
561            if ( $show_in_web === false && $show_in_exports === false ) {
562                $post_status = 'draft';
563            } elseif ( $show_in_web === true && $show_in_exports === false ) {
564                $post_status = 'web-only';
565            } elseif ( $show_in_web === false && $show_in_exports === true ) {
566                $post_status = 'private';
567            } elseif ( $show_in_web === true && $show_in_exports === true ) {
568                $post_status = 'publish';
569            }
570        }
571
572        // Title
573        if ( $show_title ) {
574            update_post_meta( $post_id, 'pb_show_title', 'on' );
575        } else {
576            delete_post_meta( $post_id, 'pb_show_title' );
577        }
578
579        // Password
580        if ( $show_in_web === false || $require_password === false ) {
581            $post_password = null; // Clear the password
582        } else {
583            $post_password = ! empty( $post->post_password ) ? $post->post_password : $_POST['post_password'] ?? ''; // @codingStandardsIgnoreLine
584        }
585
586        wp_update_post(
587            [
588                'ID' => $post_id,
589                'post_status' => $post_status ?? 'draft',
590                'post_password' => $post_password,
591            ]
592        );
593        $recursion = false;
594    }
595}
